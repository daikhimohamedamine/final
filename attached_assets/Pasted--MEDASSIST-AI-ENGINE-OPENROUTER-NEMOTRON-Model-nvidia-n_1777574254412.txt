## IMPORTANT: HOW THIS MODEL HANDLES TOOL USE

`nvidia/nemotron-3-nano-omni-30b-a3b-reasoning:free` is a **reasoning model**. It does NOT support native OpenAI function calling (`tools` array). It reasons through problems in `<think>` blocks before responding.

**Strategy**: Implement tool use via **structured JSON prompting** inside the system prompt. The model will output tool call instructions as JSON inside its response. The engine parses them, executes the real DB handler, injects results back, and re-calls the model. This is a manual agentic loop — no native tool_use protocol.

This is the correct and only viable approach for this model. Do not attempt to pass a `tools` array to OpenRouter for this model — it will be ignored or cause errors.

## TOOL PARSER — `src/ai/tool-parser.ts`

The model outputs tool calls as fenced JSON blocks. This parser extracts them.

```typescript
export interface ParsedToolCall {
  tool: string
  input: Record<string, unknown>
}

const TOOL_CALL_REGEX = /```tool_call\s*([\s\S]*?)```/

export function extractToolCall(text: string): ParsedToolCall | null {
  // Strip <think>...</think> blocks first — reasoning model includes these
  const stripped = text.replace(/<think>[\s\S]*?<\/think>/g, '').trim()

  const match = stripped.match(TOOL_CALL_REGEX)
  if (!match) return null

  try {
    const parsed = JSON.parse(match[1].trim())
    if (!parsed.tool || !parsed.input) return null
    return parsed as ParsedToolCall
  } catch {
    return null
  }
}

export function stripToolCallBlock(text: string): string {
  return text
    .replace(/<think>[\s\S]*?<\/think>/g, '')
    .replace(TOOL_CALL_REGEX, '')
    .trim()
}

export function extractThinkingBlock(text: string): string | null {
  const match = text.match(/<think>([\s\S]*?)<\/think>/)
  return match ? match[1].trim() : null
}
```

```typescript
  get_patient_history: handleGetPatientHistory,
  search_medical_library: handleSearchMedicalLibrary,
  generate_prescription: handleGeneratePrescription,
  recommend_doctor: handleRecommendDoctor,
  check_drug_interactions: handleCheckDrugInteractions,

export async function executeTool(
  toolName: string,
  input: Record<string, unknown>,
  user: { id: string; role: string }
): Promise<{ success: boolean; data?: unknown; error?: string; message?: string }> {
  const handler = TOOL_HANDLERS[toolName]
  if (!handler) {
    return { success: false, error: 'UNKNOWN_TOOL', message: `Tool '${toolName}' does not exist.` }
  }
  try {
    return await handler(input, user)
  } catch (err: any) {
    return { success: false, error: 'HANDLER_ERROR', message: err.message ?? 'Tool execution failed.' }
  }
}
```

```typescript
export async function handleGeneratePrescription(input: any, user: any) {
  // 1. Role guard
  if (user.role !== 'doctor') {
    return { success: false, error: 'FORBIDDEN', message: 'Prescription generation requires doctor role.' }
  }
  // 2. Run drug interaction check first
  const interactionCheck = await handleCheckDrugInteractions(
    { new_drugs: input.medications.map((m: any) => m.name), patient_id: input.patient_id },
    user
  )
  if (!interactionCheck.success) return interactionCheck
  const interactions = interactionCheck.data as any
  if (!interactions.safe_to_prescribe) {
    return {
      success: false,
      error: 'INTERACTION_BLOCK',
      message: 'Prescription blocked due to major drug interactions or allergy conflicts.',
      data: interactions
    }
  }
  // 3. Insert into prescriptions table
  // 4. Generate PDF with pdfkit (install if not present: npm install pdfkit)
  //    Template sections: platform header, doctor info, patient info, medication table, diagnosis, validity date, QR code block
  // 5. Generate QR code with 'qrcode' package pointing to /verify/prescription/:id
  // 6. Save PDF to ./uploads/prescriptions/:id.pdf
  // 7. Return prescription record
  return { success: true, data: { prescription_id: '', pdf_url: '', interaction_warnings: interactions.interactions } }
}
```

```typescript
// Bundled critical interaction pairs (extend to 100+ pairs)
const KNOWN_INTERACTIONS: Array<{ drugs: [string, string]; severity: 'MAJOR' | 'MODERATE' | 'MINOR'; description: string; recommendation: string }> = [
  { drugs: ['warfarin', 'aspirin'], severity: 'MAJOR', description: 'Increased bleeding risk.', recommendation: 'Avoid combination or monitor INR closely.' },
  { drugs: ['ssri', 'maoi'], severity: 'MAJOR', description: 'Risk of serotonin syndrome.', recommendation: 'Contraindicated. Allow 14-day washout.' },
  { drugs: ['metformin', 'contrast dye'], severity: 'MAJOR', description: 'Risk of lactic acidosis.', recommendation: 'Hold metformin 48h before and after contrast.' },
  { drugs: ['digoxin', 'amiodarone'], severity: 'MAJOR', description: 'Digoxin toxicity risk.', recommendation: 'Reduce digoxin dose by 50%, monitor levels.' },
  { drugs: ['sildenafil', 'nitrates'], severity: 'MAJOR', description: 'Severe hypotension.', recommendation: 'Absolutely contraindicated.' },
  { drugs: ['lithium', 'nsaid'], severity: 'MODERATE', description: 'NSAIDs reduce lithium clearance.', recommendation: 'Monitor lithium levels closely.' },
  { drugs: ['clopidogrel', 'omeprazole'], severity: 'MODERATE', description: 'Reduced antiplatelet effect.', recommendation: 'Use pantoprazole instead.' },
  // ... add more pairs
]

export async function handleCheckDrugInteractions(input: any, user: any) {
  // 1. Fetch patient current_medications and allergies from DB
  // 2. Build full drug list = new_drugs + existing_medications + additional_drugs
  // 3. Check every pair in full drug list against KNOWN_INTERACTIONS
  // 4. Check new_drugs against patient allergies (name match + common cross-reactivity rules)
  //    e.g. penicillin allergy → flag cephalosporins as MODERATE risk
  // 5. Try FDA API if accessible: https://api.fda.gov/drug/event.json?search=patient.drug.medicinalproduct:DRUGNAME
  const interactions: any[] = []
  const allergyConflicts: any[] = []

  const allDrugs = [
    ...(input.new_drugs ?? []),
    ...(input.additional_drugs ?? []),
    // + fetched patient medications from DB
  ].map((d: string) => d.toLowerCase())

  for (const pair of KNOWN_INTERACTIONS) {
    const [a, b] = pair.drugs
    const aPresent = allDrugs.some(d => d.includes(a))
    const bPresent = allDrugs.some(d => d.includes(b))
    if (aPresent && bPresent) {
      interactions.push({ drug_a: a, drug_b: b, ...pair })
    }
  }

  const hasMajor = interactions.some(i => i.severity === 'MAJOR') || allergyConflicts.some(c => c.severity === 'MAJOR')

  return {
    success: true,
    data: {
      interactions,
      allergy_conflicts: allergyConflicts,
      safe_to_prescribe: !hasMajor,
      summary: hasMajor ? 'UNSAFE: Major interactions or allergy conflicts detected.' : 'No major interactions found.'
    }
  }
}
```

## CORE ENGINE — `src/ai/engine.ts`

This is the agentic reasoning loop. It handles streaming, parses tool calls from model output, executes handlers, and re-injects results.

```typescript
import OpenAI from 'openai'
import { openrouter, AI_MODEL } from './client'
import { buildSystemPrompt } from './system-prompt'
import { extractToolCall, stripToolCallBlock, extractThinkingBlock } from './tool-parser'
import { executeTool } from './tools/index'
import { memoryManager } from './memory/index'

export interface AITurnParams {
  sessionId: string
  userMessage: string
  user: { id: string; role: string; name: string }
  patientContextId?: string
  onDelta: (text: string) => void
  onThinking: (thought: string) => void       // stream reasoning to client (optional UI use)
  onToolStart: (toolName: string) => void
  onToolEnd: (toolName: string, result: unknown) => void
  onDone: (usage: { input_tokens: number; output_tokens: number; reasoning_tokens?: number }) => void
  onError: (error: Error) => void
}

const MAX_TOOL_ITERATIONS = 6  // prevent infinite loops

export async function runAITurn(params: AITurnParams): Promise<void> {
  const { sessionId, userMessage, user } = params

  try {
    // 1. Load history
    let history = await memoryManager.load(sessionId)
    history = await memoryManager.summarizeIfNeeded(history, 20)

    // 2. Append user message
    history.push({ role: 'user', content: userMessage })

    let iterations = 0
    let finalUsage = { input_tokens: 0, output_tokens: 0, reasoning_tokens: 0 }

    // 3. Agentic tool loop
    while (iterations < MAX_TOOL_ITERATIONS) {
      iterations++

      // 4. Call OpenRouter with streaming
      const stream = await openrouter.chat.completions.create({
        model: AI_MODEL,
        messages: [
          { role: 'system', content: buildSystemPrompt(user) },
          ...history
        ],
        max_tokens: parseInt(process.env.AI_MAX_TOKENS ?? '2048'),
        stream: true,
      })

      let fullResponse = ''
      let inputTokens = 0
      let outputTokens = 0
      let reasoningTokens = 0

      // 5. Collect streamed response
      for await (const chunk of stream) {
        const delta = chunk.choices[0]?.delta?.content ?? ''
        if (delta) {
          fullResponse += delta
          // Extract and stream thinking separately — do not send raw <think> to user
          // Buffer until </think> closed, then stream only the clean content
        }
        // Usage comes in final chunk for nemotron
        if (chunk.usage) {
          inputTokens = chunk.usage.prompt_tokens ?? 0
          outputTokens = chunk.usage.completion_tokens ?? 0
          reasoningTokens = (chunk.usage as any).reasoningTokens ?? 0
        }
      }

      finalUsage = { input_tokens: inputTokens, output_tokens: outputTokens, reasoning_tokens: reasoningTokens }

      // 6. Extract thinking block (do not expose to user unless debug mode)
      const thinking = extractThinkingBlock(fullResponse)
      if (thinking) params.onThinking(thinking)

      // 7. Check for tool call in response
      const toolCall = extractToolCall(fullResponse)

      if (!toolCall) {
        // No tool call — this is the final answer. Stream clean text to user.
        const cleanResponse = stripToolCallBlock(fullResponse)
        params.onDelta(cleanResponse)

        // Persist to history
        history.push({ role: 'assistant', content: cleanResponse })
        break
      }

      // 8. Tool call detected — execute it
      params.onToolStart(toolCall.tool)

      const toolResult = await executeTool(toolCall.tool, toolCall.input, user)

      params.onToolEnd(toolCall.tool, toolResult)

      // 9. Inject tool call + result into history so model can continue
      history.push({
        role: 'assistant',
        content: stripToolCallBlock(fullResponse) + `\n\`\`\`tool_call\n${JSON.stringify({ tool: toolCall.tool, input: toolCall.input }, null, 2)}\n\`\`\``
      })
      history.push({
        role: 'user',  // OpenRouter/OpenAI format: tool results injected as user messages
        content: `[TOOL RESULT: ${toolCall.tool}]\n${JSON.stringify(toolResult, null, 2)}\n[/TOOL RESULT]\nContinue your response based on this data.`
      })
      // Loop continues — model will now respond using the tool result
    }

    if (iterations >= MAX_TOOL_ITERATIONS) {
      params.onDelta('\n\n*Maximum tool call depth reached. Please rephrase or break your question into parts.*')
    }

    // 10. Save updated history
    await memoryManager.save(sessionId, history)

    // 11. Persist to DB chat_messages table (full turn)
    // Insert user message + assistant response + tool metadata

    params.onDone(finalUsage)

  } catch (err: any) {
    params.onError(err)
  }
}
```

## MEMORY MANAGER — `src/ai/memory/index.ts`

```typescript
import { openrouter, AI_MODEL } from '../client'

type Message = { role: string; content: string }

// Redis client — use existing client from codebase
// If no Redis: fallback to in-process Map
const store = new Map<string, { messages: Message[]; expiry: number }>()
const TTL = parseInt(process.env.AI_SESSION_TTL_SECONDS ?? '7200') * 1000

export const memoryManager = {
  async load(sessionId: string): Promise<Message[]> {
    // Try Redis first using existing redisClient from codebase
    // Fallback: in-memory store
    const entry = store.get(sessionId)
    if (!entry || Date.now() > entry.expiry) return []
    return entry.messages
  },

  async save(sessionId: string, messages: Message[]): Promise<void> {
    store.set(sessionId, { messages, expiry: Date.now() + TTL })
    // Also persist to Redis if available
  },

  async clear(sessionId: string): Promise<void> {
    store.delete(sessionId)
  },

  async summarizeIfNeeded(messages: Message[], threshold: number): Promise<Message[]> {
    if (messages.length <= threshold) return messages

    const toSummarize = messages.slice(0, Math.floor(threshold / 2))
    const toKeep = messages.slice(Math.floor(threshold / 2))

    try {
      const summaryResponse = await openrouter.chat.completions.create({
        model: AI_MODEL,
        messages: [
          {
            role: 'user',
            content: `Summarize this medical conversation. Preserve: all clinical data mentioned, medications, diagnoses, decisions made, and patient concerns. Be concise but clinically complete.\n\n${toSummarize.map(m => `${m.role}: ${m.content}`).join('\n')}`
          }
        ],
        max_tokens: 600,
        stream: false,
      })

      const summary = (summaryResponse.choices[0]?.message?.content ?? '').replace(/<think>[\s\S]*?<\/think>/g, '').trim()

      return [
        { role: 'user', content: `[Previous conversation summary]: ${summary}` },
        { role: 'assistant', content: 'Understood. I have the context from our previous discussion.' },
        ...toKeep
      ]
    } catch {
      // If summarization fails, just trim the oldest messages
      return messages.slice(-threshold)
    }
  }
}
```

```typescript
// POST /api/ai/chat
// Body: { sessionId: string, message: string, patientContextId?: string }
// Auth: required
// Response: text/event-stream

async function aiChatHandler(req, reply) {
  const user = req.user // from existing auth middleware
  const { sessionId, message, patientContextId } = req.body

  reply.raw.setHeader('Content-Type', 'text/event-stream')
  reply.raw.setHeader('Cache-Control', 'no-cache')
  reply.raw.setHeader('Connection', 'keep-alive')
  reply.raw.flushHeaders()

  const send = (data: object) => {
    reply.raw.write(`data: ${JSON.stringify(data)}\n\n`)
  }

  await runAITurn({
    sessionId,
    userMessage: message,
    user: { id: user.id, role: user.role, name: user.full_name },
    patientContextId,
    onDelta: (text) => send({ type: 'delta', text }),
    onThinking: (thought) => send({ type: 'thinking', thought }),  // optional — hide in production UI
    onToolStart: (tool) => send({ type: 'tool_start', tool }),
    onToolEnd: (tool, result) => send({ type: 'tool_end', tool }),
    onDone: (usage) => {
      send({ type: 'done', usage })
      reply.raw.end()
    },
    onError: (err) => {
      send({ type: 'error', message: err.message })
      reply.raw.end()
    }
  })
}
```
